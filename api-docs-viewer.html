üí° Quick Start Examples

üöÄ Basic Usage

# Discover masters
curl -H "X-API-Key: your-api-key" http://localhost:3000/api/v1/masters

# Connect to a master
curl -X POST -H "X-API-Key: your-api-key" \
     -H "Content-Type: application/json" \
     -d '{"deviceName": "COM4"}' \
     http://localhost:3000/api/v1/masters/connect

# List devices
curl -H "X-API-Key: your-api-key" http://localhost:3000/api/v1/devices

# Read process data
curl -H "X-API-Key: your-api-key" \
     http://localhost:3000/api/v1/data/1/1/process
